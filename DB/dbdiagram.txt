Table User {
  id integer  [primary key]
  e_mail integer
  password_hash varchar
  name varchar
  surname varchar
  title varchar
  specialization varchar
  practice_start_year integer

}

Table Group {
  id integer  [primary key] 
  name varchar 
  description varchar

}

Table Task {
  id integer  [primary key]
  id_group integer
  max_samples_for_user integer
  name varchar
  description varchar
  type varchar

}

Table Membership{
  id_user integer
  id_group integer
  role varchar
}


Table Sample
{
  id integer [ primary key ]
  id_task integer
  id_user integer
  path varchar
  format varchar

}

Table Examination{
   id integer  [primary key]
   id_user integer
   id_sample integer
   to_further_verification bool
   bad_quality bool

}

Table Label{
  id integer [primary key]
  id_task integer
  name varchar
  description varchar

}

Table BBox{
  id integer [primary key]
  id_examination integer
  label_id integer
  comment varchar
  x1 float
  y1 float
  x2 float
  y2 float  
}
