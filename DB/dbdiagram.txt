Table User {
  id integer  [primary key]
  e_mail integer
  password_hash varchar
  name varchar
  surname varchar
  title varchar
  specialization varchar
  practice_start_year integer

}

Table Group {
  id integer  [primary key] 
  name varchar 
  description varchar

}

Table Set {
  id integer  [primary key]
  id_group integer  
  name varchar
  description varchar
  type varchar
  package_size integer

}

Table Package{
  id integer  [primary key]
  id_set integer
  id_user integer
  is_ready bool
}

Table Membership{
  id_user integer
  id_group integer
  role varchar
}


Table Sample
{
  id integer [ primary key ]
  id_package integer
 
  path varchar
  format varchar

}

Table Examination{
   id integer  [primary key]
   id_user integer
   id_sample integer
   to_further_verification bool
   bad_quality bool

}

Table Label{
  id integer [primary key]
  id_set integer
  name varchar
  description varchar

}

Table BBox{
  id integer [primary key]
  id_examination integer
  label_id integer
  comment varchar
  x int
  y int
  width int
  height int
}
